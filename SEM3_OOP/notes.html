// 1. Account class with credit, debit, transferTo methods
class Account {
    private double balance;
    private String accNumber;

    public Account(String accNumber) {
        this.accNumber = accNumber;
        this.balance = 0;
    }

    public void credit(double amount) {
        if(amount > 0) balance += amount;
    }

    public void debit(double amount) {
        if(amount > 0 && balance >= amount) balance -= amount;
        else System.out.println("Insufficient balance");
    }

    public void transferTo(Account other, double amount) {
        if(amount > 0 && balance >= amount) {
            this.debit(amount);
            other.credit(amount);
        } else System.out.println("Transfer failed");
    }

    public void display() {
        System.out.printf("Account: %s, Balance: %.2f\n", accNumber, balance);
    }
}

        a1.display();
        a2.display();
    }
}

// 2. String concatenation with trimming longer string
public class StringConcat {
    public static String concatTrimmed(String s1, String s2) {
        int len = Math.min(s1.length(), s2.length());
        return s1.substring(s1.length()-len) + s2.substring(s2.length()-len);
    }

    public static void main(String[] args) {
        System.out.println(concatTrimmed("Hello", "Hi")); // loHi
    }
}

// 3. Find 2nd largest element in int array
public class SecondLargest {
    public static int findSecondLargest(int[] arr) {
        int max = Integer.MIN_VALUE, secondMax = Integer.MIN_VALUE;
        for(int val : arr) {
            if(val > max) {
                secondMax = max;
                max = val;
            } else if(val > secondMax && val < max) {
                secondMax = val;
            }
        }
        return secondMax;
    }

    public static void main(String[] args) {
        int[] arr = {12, 35, 1, 10, 34, 1};
        System.out.println("Second Largest: " + findSecondLargest(arr)); // 34
    }
}

// 4. Article class with loss percentage calculation
class Article {
    String name;
    float costPrice, sellingPrice, lossPercent;

    void calc_loss_per() {
        lossPercent = ((costPrice - sellingPrice) / costPrice) * 100;
    }

    static Article find_high_loss(Article[] articles) {
        Article maxLossArticle = articles[0];
        for(Article a : articles) {
            if(a.lossPercent > maxLossArticle.lossPercent) maxLossArticle = a;
        }
        return maxLossArticle;
    }
}

// 8. Employee attendance with ArrayList and search/filter
class Employee {
    int empId, daysPresent;
    String name;

    Employee(int empId, String name, int daysPresent) {
        this.empId = empId; this.name = name; this.daysPresent = daysPresent;
    }

    void display() {
        System.out.printf("ID: %d, Name: %s, Days Present: %d\n", empId, name, daysPresent);
    }
}

class EmployeeManager {
    public static void main(String[] args) {
        ArrayList emps = new ArrayList<>();
        emps.add(new Employee(101, "John", 22));
        emps.add(new Employee(102, "Sara", 18));
        emps.add(new Employee(103, "Mike", 25));
        emps.add(new Employee(104, "Anna", 15));
        emps.add(new Employee(105, "Tom", 21));

        System.out.println("All Employees:");
        emps.forEach(Employee::display);

        Scanner sc = new Scanner(System.in);
        System.out.print("Enter empId to search: ");
        int id = sc.nextInt();

        boolean found = false;
        for(Employee e : emps) {
            if(e.empId == id) {
                e.display();
                found = true;
                break;
            }
        }
        if(!found) System.out.println("Employee not found");

        System.out.println("Employees with attendance < 20:");
        for(Employee e : emps) {
            if(e.daysPresent < 20) e.display();
        }
    }
}

// 12. Comparable interface to sort books by price
import java.util.*;

class Book implements Comparable {
    String title, author;
    double price;

    Book(String t, String a, double p) {
        title = t; author = a; price = p;
    }

    public int compareTo(Book b) {
        return Double.compare(this.price, b.price);
    }

    public String toString() {
        return "Title: " + title + " Author: " + author + " Price: " + price;
    }

    public static void main(String[] args) {
        ArrayList books = new ArrayList<>();
        books.add(new Book("Java Basics", "John Doe", 350.0));
        books.add(new Book("Data Structures", "Smith", 450.0));
        books.add(new Book("Advanced Java", "Jane", 550.0));

        Collections.sort(books);
        System.out.println("Books sorted by price:");
        books.forEach(System.out::println);
    }
}

// 13. Comparator interface example to sort Employees by name and age
import java.util.*;

class EmployeeC {
    int id, age;
    String name;

    EmployeeC(int id, String name, int age) {
        this.id = id; this.name = name; this.age = age;
    }

    public String toString() {
        return id + " " + name + " " + age;
    }
}

class NameComparator implements Comparator {
    public int compare(EmployeeC e1, EmployeeC e2) {
        return e1.name.compareTo(e2.name);
    }
}

class AgeComparator implements Comparator {
    public int compare(EmployeeC e1, EmployeeC e2) {
        return Integer.compare(e1.age, e2.age);
    }
}

class EmployeeSortDemo {
    public static void main(String[] args) {
        ArrayList list = new ArrayList<>();
        list.add(new EmployeeC(1, "John", 25));
        list.add(new EmployeeC(2, "Anna", 22));
        list.add(new EmployeeC(3, "Mike", 28));

        Collections.sort(list, new NameComparator());
        System.out.println("Sorted by name:");
        list.forEach(System.out::println);

        Collections.sort(list, new AgeComparator());
        System.out.println("Sorted by age:");
        list.forEach(System.out::println);
    }
}

// 14. Exception handling example with try-catch
import java.util.Scanner;

class SafeCalculator {
    int divide(int a, int b) {
        try {
            return a / b;
        } catch(ArithmeticException e) {
            System.out.println("Error: Cannot divide by zero.");
            return 0;  // default value
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Numerator: ");
        int a = sc.nextInt();
        System.out.print("Denominator: ");
        int b = sc.nextInt();

        SafeCalculator calc = new SafeCalculator();
        int result = calc.divide(a, b);
        if(b != 0) System.out.println("Result: " + result);
    }
}

// 15. throw and throws example with FileReaderUtility
import java.io.IOException;

class FileReaderUtility {
    void readFile(String fileName) throws IOException {
        if(fileName.isEmpty()) throw new IllegalArgumentException("Filename cannot be empty.");
        if(fileName.equals("missing.txt")) throw new IOException("File not found.");
        System.out.println("Reading file: " + fileName);
    }

    public static void main(String[] args) {
        String[] files = {"data.txt", "", "missing.txt"};
        FileReaderUtility fru = new FileReaderUtility();

        for(String file : files) {
            try {
                fru.readFile(file);
            } catch(IllegalArgumentException | IOException e) {
                System.out.println("Error: " + e.getMessage());
            }
        }
    }
}
